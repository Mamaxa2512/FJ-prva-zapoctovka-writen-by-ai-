from enum import Enum, auto
from collections import deque

# Definícia stavov automatu
class State(Enum):
    q0  = auto()
    q1  = auto()
    q2  = auto()
    q3  = auto()
    q4  = auto()
    q5  = auto()
    q6  = auto()
    q7  = auto()
    q8  = auto()
    q9  = auto()
    q10 = auto()

class NKA:
    def __init__(self, transition_table: dict,
                 accepted_states: set, init_state: State):
        self.transition_table  = transition_table
        self.accepted_states    = accepted_states
        self.init_state         = init_state
        self.queue              = None

    def expand(self, state: State, string: str) -> None:
        # Prechod podľa aktuálneho symbolu
        if string != "" and (state, string[0]) in self.transition_table:
            for s in self.transition_table[(state, string[0])]:
                self.queue.append((s, string[1:]))

        # Epsilon-prechody (bez čítania symbolu)
        if (state, "") in self.transition_table:
            for s in self.transition_table[(state, "")]:
                self.queue.append((s, string))   # reťazec sa neskracuje!

    def check(self, string: str) -> bool:
        # BFS — prehľadávanie do šírky
        self.queue = deque()
        self.queue.append((self.init_state, string))

        while self.queue:
            state, rest = self.queue.popleft()  # FIFO!

            if state in self.accepted_states and rest == "":
                return True

            self.expand(state, rest)

        return False


# Prechodová tabuľka
transition_table = {
    (State.q0, ''): {State.q1, State.q3},
    (State.q1, '0'): {State.q2},
    (State.q3, '1'): {State.q4},
    (State.q4, ''): {State.q5},
    (State.q5, ''): {State.q6},
    (State.q6, ''): {State.q7, State.q9},
    (State.q7, '0'): {State.q8},
    (State.q8, ''): {State.q6},
    (State.q9, '1'): {State.q10},
    (State.q10, ''): {State.q6},
}
accepted_states = {State.q2, State.q5, State.q8, State.q10}

nka = NKA(transition_table, accepted_states, State.q0)

s = input("Zadaj reťazec> ")
while s != "quit":
    ok = nka.check(s)
    print(f"'{s}' ", "JE" if ok else "NIE JE", "akceptovaný")
    s = input("Zadaj reťazec> ")

from enum import Enum, auto

class State(Enum):
    q0=auto(); q1=auto(); q2=auto(); q3=auto()
    q4=auto(); q5=auto(); q6=auto(); q7=auto()
    q8=auto(); q9=auto(); q10=auto()

# Každý stav = funkcia
# Prechod = volanie inej funkcie
# Epsilon = volanie bez skrátenia reťazca

def q0(s: str) -> bool:
    # epsilon-prechody do q1 a q3 → or = nedeterminizmus
    return q3(s) or q1(s)

def q1(s: str) -> bool:
    # čakáme symbol '0'
    if len(s) > 0 and s[0] == '0':
        return q2(s[1:])  # skrátime reťazec
    return False

def q2(s: str) -> bool:
    # akceptačný stav — reťazec musí byť prázdny
    return len(s) == 0

def q3(s: str) -> bool:
    # čakáme symbol '1'
    if len(s) > 0 and s[0] == '1':
        return q4(s[1:])
    return False

def q4(s: str) -> bool:
    # akceptačný + epsilon do q5
    if len(s) == 0:
        return True
    return q5(s)    # epsilon — s sa neskracuje

def q5(s: str) -> bool:
    # akceptačný + epsilon do q6
    if len(s) == 0:
        return True
    return q6(s)    # epsilon

def q6(s: str) -> bool:
    # rozvetvenie: q9 (pre '1') alebo q7 (pre '0')
    return q9(s) or q7(s)

def q7(s: str) -> bool:
    # čakáme '0'
    if len(s) > 0 and s[0] == '0':
        return q8(s[1:])
    return False

def q8(s: str) -> bool:
    # akceptačný + epsilon späť do q6 = cyklus!
    if len(s) == 0:
        return True
    return q6(s)    # rekurzívny skok

def q9(s: str) -> bool:
    # čakáme '1'
    if len(s) > 0 and s[0] == '1':
        return q10(s[1:])
    return False

def q10(s: str) -> bool:
    # akceptačný + epsilon späť do q6 = cyklus!
    if len(s) == 0:
        return True
    return q6(s)

# Spustenie — q0 je vstupný bod
s = input("Zadaj reťazec> ")
while s != "quit":
    print("JE" if q0(s) else "NIE JE", "akceptovaný")

from enum import Enum, auto

# DKA: rozpoznáva reťazce začínajúce 'a' a končiace 'b'
# napr: ab, aab, axb, aaab — ÁNO
#       ba, a, b, ""       — NIE

class State(Enum):
    q0 = auto()  # počiatočný
    q1 = auto()  # prečítali sme 'a', čakáme ďalej
    q2 = auto()  # akceptačný — posledný symbol bol 'b'
    q3 = auto()  # mŕtvy stav — prvý symbol nebol 'a'

class DKA:
    def __init__(self, transition_table: dict,
                 accepted_states: set, init_state: State):
        self.transition_table = transition_table
        self.accepted_states  = accepted_states
        self.init_state       = init_state

    def check(self, string: str) -> bool:
        # začíname v počiatočnom stave
        state = self.init_state

        # čítame symbol po symbole — deterministicky
        for symbol in string:
            key = (state, symbol)
            if key in self.transition_table:
                state = self.transition_table[key]
            else:
                return False  # neexistujúci prechod = odmietnutie

        # akceptujeme ak sme v akceptačnom stave
        return state in self.accepted_states


# Prechodová tabuľka — každý (stav, symbol) → práve 1 stav
transition_table = {
    (State.q0, 'a'): State.q1,   # prečítali sme 'a'
    (State.q1, 'b'): State.q2,   # prečítali sme 'b' → akceptačný
    (State.q1, 'a'): State.q1,   # ďalšie 'a' → zostávame
    (State.q2, 'b'): State.q2,   # ďalšie 'b' → zostávame
    (State.q2, 'a'): State.q1,   # 'a' po 'b' → späť na q1
    # ostatné symboly → return False (mŕtvy stav implicitný)
}
accepted_states = {State.q2}

dka = DKA(transition_table, accepted_states, State.q0)

s = input("Zadaj reťazec> ")
while s != "quit":
    ok = dka.check(s)
    print("JE" if ok else "NIE JE", "akceptovaný")
    s = input("Zadaj reťazec> ")

from enum import Enum, auto

# DKA rekurzívne — každý stav = funkcia
# Hĺbka rekurzie = dĺžka reťazca (bezpečné!)

class State(Enum):
    q0 = auto()
    q1 = auto()
    q2 = auto()
    q3 = auto()

def q0(s: str) -> bool:
    # počiatočný stav — prvý symbol musí byť 'a'
    if len(s) == 0:
        return False          # prázdny reťazec — nie
    if s[0] == 'a':
        return q1(s[1:])     # 'a' prečítané → q1
    return q3(s[1:])         # iný symbol → mŕtvy stav

def q1(s: str) -> bool:
    # prečítali sme 'a', čakáme ďalej
    if len(s) == 0:
        return False          # skončili v q1 — nie je akceptačný
    if s[0] == 'b':
        return q2(s[1:])     # 'b' → akceptačný stav
    return q1(s[1:])         # čokoľvek iné → zostávame v q1

def q2(s: str) -> bool:
    # akceptačný stav — posledný prečítaný symbol bol 'b'
    if len(s) == 0:
        return True           # reťazec prečítaný → AKCEPTOVANÝ!
    if s[0] == 'b':
        return q2(s[1:])     # ďalšie 'b' → zostávame
    return q1(s[1:])         # 'a' alebo iné → späť na q1

def q3(s: str) -> bool:
    # mŕtvy stav — odtiaľto niet úniku
    return False

# Príklady priebehu:
# q0("aab") → q1("ab") → q1("b") → q2("") → True
# q0("ba")  → q3("a")             → False
# q0("ab")  → q1("b") → q2("")    → True

s = input("Zadaj reťazec> ")
while s != "quit":
    print("JE" if q0(s) else "NIE JE", "akceptovaný")
    s = input("Zadaj reťazec> ")

Odpoveď
0
X⁺ ⊆ X*, teda X⁺ − X* = ∅ → mohutnosť 0.
04

5
L₁·L₂ = {ε, a, b, aa, ab} — zreťazíme každý prvok L₁ s každým prvkom L₂ a odstrânime duplicity. |L₁·L₂| = 5.
05
Nech L₁ = {ε, aaa, ab, ba, bba} a L₂ = {ε}. Vymenujte všetky prvky jazyka ktorý vznikne ako ľavý kvocient jazyka L₁ podľa jazyka L₂. Pre prázdne slovo použite zápis eps.
Odpoveď
eps aaa ab ba bba
Ľavý kvocient L₁/L₂ = { w | ∃u∈L₂: uw∈L₁ }. Keďže L₂ = {ε}, každé slovo z L₁ zostáva nezmenené — odstraňujeme prázdny prefix.
